import sys
import tkinter as tk
from tkinter import ttk,messagebox, Menu


ventana=tk.Tk();
ventana.geometry('600x300')
ventana.title('PROYECTO')

def salir():
    ventana.quit()
    ventana.destroy()
    sys.exit()

def crear():

    ventana_nueva1 = tk.Toplevel()
    ventana_nueva1.title("Ventana secundaria")
    ventana_nueva1.geometry('300x200')


def crear_menu():
    menu_principal=Menu(ventana)
    submenu_archivo= Menu(menu_principal,tearoff=0)

    menu_principal.add_cascade(menu=submenu_archivo,label='Archivo')
    ventana.config(menu=menu_principal)
    menu_principal.add_cascade( label='Ayuda', command=crear)

    submenu_archivo.add_command(label='Salir',command=salir)

crear_menu()
